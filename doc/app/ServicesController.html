def check_user_status_and_redirect user
  if user.status == USER_STATUS_DELETED
    flash[:notice] = t("devise.failure.user_deleted")
    redirect_to new_user_session_path            
  else
    sign_in_and_redirect(:user, user)      
  end 
end

def create
  omniauth = request.env['omniauth.auth']

  if omniauth && omniauth['provider'] =='facebook' || omniauth['provider'] == 'linkedin' || omniauth['provider'] == 'google'

    # new_provider
    if Service.where(:uid => omniauth['uid'], :provider => omniauth['provider']).empty?

      # user_present_for_new_provider
      if (user = User.find_by_email(omniauth['info']['email'])).present?
        user.services << Service.new(:provider => omniauth['provider'], :uemail => omniauth['info']['email'], :uid => omniauth['uid'], :uname => omniauth['info']['name']) 
        user.save
        check_user_status_and_redirect user
     
      # no_user_for_new_provider         
      else
        if omniauth['provider'] == 'google'
          name = omniauth['info']['name'].split(' ')
          user = User.new(:first_name => name.first, :last_name => name.last, :email => omniauth['info']['email'], :user_type => USER_TYPE_CONTACT, :status => USER_STATUS_PENDING, :password => 'password')            
        else
          user = User.new(:first_name => omniauth['info']['first_name'], :last_name => omniauth['info']['last_name'], :email => omniauth['info']['email'], :user_type => USER_TYPE_CONTACT, :status => USER_STATUS_PENDING, :password => 'password')
        end  
        user.services << Service.new(:provider => omniauth['provider'], :uemail => omniauth['info']['email'], :uid => omniauth['uid'], :uname => omniauth['info']['name'])
        user.save
        flash[:confirm_notice] = t("devise.failure.unconfirmed")
        check_user_status_and_redirect user
      end
      
    # provider_present_in_db
    else
      flash[:notice] = t("devise.failure.unconfirmed")
      user = Service.where(:provider => omniauth['provider'], :uid => omniauth['uid']).first.user
      check_user_status_and_redirect user
    end
    
  else         
    render :text =>  "not supported omniauth provider : " + ( omniauth ? omniauth['provider'] : " error in omniauth ") 
  end      

end
